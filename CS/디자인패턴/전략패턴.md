## 전략패턴

전략이라고 부르는 캡슐화한 알고리즘을 컨텍스트 안에서 바꿔주면서 상호 교체가 가능하게 하는 디자인 패턴

```java
public class StrategyPatternExample {

    // === Strategy (전략 인터페이스) ===
    interface PaymentStrategy {
        void pay(int amount);
    }

    static class KakaoPayPayment implements PaymentStrategy {
        private final String kakaoId;

        public KakaoPayPayment(String kakaoId) {
            this.kakaoId = kakaoId;
        }

        @Override
        public void pay(int amount) {
            System.out.println(amount + "원을 카카오페이(" + kakaoId + ")로 결제합니다.");
        }
    }

    static class NaverPayPayment implements PaymentStrategy {
        private final String naverId;

        public NaverPayPayment(String naverId) {
            this.naverId = naverId;
        }

        @Override
        public void pay(int amount) {
            System.out.println(amount + "원을 네이버페이(" + naverId + ")로 결제합니다.");
        }
    }

    // === Context (전략을 사용하는 쪽) ===
    static class PaymentService {
        private PaymentStrategy paymentStrategy; // 합성(Composition)

        // 전략을 런타임에 교체 가능
        public void setPaymentStrategy(PaymentStrategy paymentStrategy) {
            this.paymentStrategy = paymentStrategy;
        }

        public void checkout(int amount) {
            if (paymentStrategy == null) {
                System.out.println("결제 수단이 선택되지 않았습니다.");
                return;
            }
            paymentStrategy.pay(amount);
        }
    }

    // === 실행 ===
    public static void main(String[] args) {
        PaymentService service = new PaymentService();

        service.setPaymentStrategy(new KakaoPayPayment("user_kakao"));
        service.checkout(20000);

        service.setPaymentStrategy(new NaverPayPayment("user_naver"));
        service.checkout(30000);
    }
}

```
